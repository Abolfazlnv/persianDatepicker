using persianDatepicker with angularjs

{{ date }} 
<div ng-app="newsFrom" id="newsForm" ng-controller="myCtrl">
	<input type="text"  class="publish" />
	<span>{{ date }}</span>
</div>

<script>
var app = angular.module('newsFrom', []);
app.controller('myCtrl', function($scope) {	
	$scope.date = "NOT SELECTED YET!";
	$scope.setDateVal = function () {
		$scope.date = $(".publish").val();
		$scope.$apply();
	};
});	

$( ".publish" ).persianDatepicker({
	cellWidth: 35, 
	cellHeight: 30, 
	onSelect: function(){		
		angular.element($('#newsForm')).scope().setDateVal();				
	}
});
</script>
